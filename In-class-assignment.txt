public class Mains {
    public static void main(String[] args) {
        float result = division (10);
        System.out.printf("%.7f", result);
    }

    public static float division(int i) {
        if (i - 1 == 0 || i == 0) {
            return 1;
        } else {
            return (i / division(i-1));
        }
    }

}
